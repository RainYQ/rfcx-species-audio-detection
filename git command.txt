git clone https://github.com/RainYQ/rfcx-species-audio-detection.git

添加远程源
git remote add origin https://github.com/RainYQ/rfcx-species-audio-detection.git

创建并切换到新分支
git checkout -b [分支名称]

创建提交
git add *
git commit -m "提交内容"
git push -u origin [分支名称]

切换到main分支
git checkout main

查看本地分支
git branch

从远端main分支拉取代码
git pull origin main
